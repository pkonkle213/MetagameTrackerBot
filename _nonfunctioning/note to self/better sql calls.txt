basis: https://www.psycopg.org/psycopg3/docs/advanced/rows.html

import psycopg
from psycopg.rows import class_row
from dataclasses import dataclass

# 1. Define your Fish object
@dataclass
class Fish: <- Note the definition of a class
    color: str
    length: float
    width: float

# 2. Connect and use the class_row factory
conn_info = "dbname=test user=postgres password=secret"

with psycopg.connect(conn_info) as conn:
    # Set the row_factory to your Fish class
    with conn.cursor(row_factory=class_row(Fish)) as cur: <- Note the usage of said class
        cur.execute("SELECT color, length, width FROM fish WHERE id = %s", (1,))

        # This returns a Fish object automatically
        my_fish = cur.fetchone()

        print(f"My {my_fish.color} fish is {my_fish.length}cm long.") <- Items are automatically put into the class
